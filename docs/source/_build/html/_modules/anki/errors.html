from typing import Any



[docs]class AnkiError(Exception):
    def __init__(self, type, **data) -> None:
        super().__init__()
        self.type = type
        self.data = data

    def __str__(self) -> Any:
        m = self.type
        if self.data:
            m += ": %s" % repr(self.data)
        return m

        super().__init__()
